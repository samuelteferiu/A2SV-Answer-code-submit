class Solution:
    def maximumElementAfterDecrementingAndRearranging(self, arr: List[int]) -> int:
        arr=sorted(arr)
        maxs=0
        if arr[0]==1:
            maxs=1
            for i in range(1,len(arr)):
                if arr[i]-maxs>1:
                    maxs+=1
                else:
                    maxs=max(maxs,arr[i])
            return maxs
        else:
            arr[0]=1
            maxs=1
            for i in range(1,len(arr)):
                if arr[i]-maxs>1:
                    maxs+=1
                else:
                    maxs=max(maxs,arr[i])
            return maxs
