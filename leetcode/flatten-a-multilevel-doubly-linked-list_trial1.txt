"""
# Definition for a Node.
class Node:
    def __init__(self, val, prev, next, child):
        self.val = val
        self.prev = prev
        self.next = next
        self.child = child
"""
class Solution:
    def flatten(self, head: 'Node') -> 'Node':
        def dfs(node):
            cur = node
            last = None 
            while cur:
                next_node = cur.next
                if cur.child:
                    child_head = cur.child
                    child_tail = dfs(child_head)
                    cur.next = child_head
                    child_head.prev = cur
                    cur.child = None
                    if next_node:
                        child_tail.next = next_node
                        next_node.prev = child_tail
                    
                    last = child_tail
                    cur = next_node
                else:
                    last = cur
                    cur = next_node
            
            return last
        if head:
            dfs(head)
        
        return head
