class Solution:
    def shipWithinDays(self, weights: List[int], days: int) -> int:
        
        def canhold(capacity):
            d = 1
            curr = 0
            
            for w in weights:
                if curr + w > capacity:
                    d += 1
                    curr = 0
                curr += w
            
            return d <= days
        
        l, r = max(weights), sum(weights)
        ans = r
    
        while l <= r:
            mid = (l + r) // 2
            
            if canhold(mid):
                ans = mid
                r = mid - 1   
            else:
                l = mid + 1 
        
        return ans
