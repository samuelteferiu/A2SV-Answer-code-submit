class Solution:
    def minimumRecolors(self, blocks: str, k: int) -> int:
        mins=9999999999999999
        count=Counter(blocks[:k])
        mins=min(mins,count["W"])
        l=0
        r=k
        while r<len(blocks):
            count[blocks[r]]=count.get(blocks[r], 0) + 1
            count[blocks[l]]-=1
            mins=min(mins,count["W"])
            l+=1
            r+=1
        return mins
