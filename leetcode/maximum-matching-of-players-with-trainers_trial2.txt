class Solution:
    def matchPlayersAndTrainers(self, players: List[int], trainers: List[int]) -> int:
        train=sorted(trainers)
        play=sorted(players)
        count=0
        left=0
        right=0
        while left<len(play) and right<len(train):
            if play[left]<=train[right]:
                count+=1
                left+=1
                right+=1
            else:
                right+=1
        return count
