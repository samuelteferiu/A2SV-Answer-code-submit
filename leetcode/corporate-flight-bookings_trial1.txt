class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        dicts = defaultdict(int)
        for start, end, val in bookings:
            dicts[start - 1] += val
            if end < n:
                dicts[end] -= val
        ans = [0] * n
        curr = 0
        for i in range(n):
            curr += dicts[i]
            ans[i] = curr
        return ans
