class Solution:
    def subarraysWithKDistinct(self, nums, k):
        return self.atMost(nums, k) - self.atMost(nums, k - 1)
    def atMost(self, nums, k):
        count = defaultdict(int)
        left = 0
        ans = 0
        for right in range(len(nums)):
            count[nums[right]] += 1

            while len(count) > k:
                count[nums[left]] -= 1
                if count[nums[left]] == 0:
                    del count[nums[left]]
                left += 1
            ans += right - left + 1
        return ans
