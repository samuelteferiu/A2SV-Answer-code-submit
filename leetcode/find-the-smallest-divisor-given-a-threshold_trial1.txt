class Solution:
    def smallestDivisor(self, nums: List[int], threshold: int) -> int:
        l=1
        r=max(nums)
        ans=r
        while l<=r:
            mid=((l+r)//2)
            sums = sum(math.ceil(i / mid) for i in nums)
            if sums>threshold:
                l=mid+1
            else:
                ans=min(ans,mid)
                r=mid-1
        return ans
