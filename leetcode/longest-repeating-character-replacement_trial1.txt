class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        count={}
        l=0
        maxs=0
        for r in range(len(s)):
            count[s[r]]=count.get(s[r], 0) + 1
            leng=r-l+1
            if leng-max(count.values())>k:
                count[s[l]]-=1
                l+=1
            else:
                maxs=max(maxs,leng)
        return maxs
