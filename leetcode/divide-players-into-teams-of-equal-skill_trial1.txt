class Solution:
    def dividePlayers(self, skill: List[int]) -> int:
        skill.sort()
        n = len(skill)
        target = (sum(skill) * 2) // n  
        ans = 0
        left, right = 0, n - 1
        
        while left < right:
            if skill[left] + skill[right] != target:
                return -1
            ans += skill[left] * skill[right]
            left += 1
            right -= 1
        
        return ans
