class DataStream:

    def __init__(self, value: int, k: int):
        self.val = value
        self.k = k
        self.count =[] 
    def consec(self, num: int) -> bool:
        if self.val==num:
            self.count.append(num)
        else:
            self.count=[]
        if len(self.count)>=self.k:
            return True
        else:
            return False
