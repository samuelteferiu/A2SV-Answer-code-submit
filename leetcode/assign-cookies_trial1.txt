class Solution:
    def findContentChildren(self, g: List[int], s: List[int]) -> int:
        right=0
        left=0
        ans=0
        g=sorted(g)
        s=sorted(s)
        if not s:
            return 0
        else:
            while right<len(g) and left<len(s):
                if g[right]<=s[left]:
                    ans+=1
                    left+=1
                    right+=1
                else:
                    left+=1
            return ans
