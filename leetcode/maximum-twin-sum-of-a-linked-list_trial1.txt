# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def pairSum(self, head: Optional[ListNode]) -> int:
        arr=[]
        cur=head
        while cur:
            arr.append(cur.val)
            cur=cur.next
        maxs=0
        l=0
        r=len(arr)-1
        while l<r:
            x=arr[r]+arr[l]
            maxs=max(maxs,x)
            l+=1
            r-=1
        return maxs
