class Solution:
    def numSubarrayProductLessThanK(self, nums: List[int], k: int) -> int:
        pre = 1
        ans = 0
        left = 0
        for right, val in enumerate(nums):
            pre *= val
            while pre >= k and left <= right:
                pre /= nums[left]   
                left += 1
            ans += right - left + 1
        return ans
