class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        top = 0
        bottom = len(matrix) - 1
        while top <= bottom:
            mid = (top + bottom) // 2
            if matrix[mid][0] <= target <= matrix[mid][-1]:
                left = 0
                right = len(matrix[mid]) - 1

                while left <= right:
                    m = (left + right) // 2
                    if matrix[mid][m] == target:
                        return True
                    elif matrix[mid][m] < target:
                        left = m + 1
                    else:
                        right = m - 1
                return False

            elif target < matrix[mid][0]:
                bottom = mid - 1
            else: 
                top = mid + 1

        return False
