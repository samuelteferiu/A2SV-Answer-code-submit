class Solution:
    def findRightInterval(self, intervals: List[List[int]]) -> List[int]:
        n = len(intervals)
        starts = []
        for i in range(n):
            starts.append((intervals[i][0], i))
        starts.sort()
        newstarts = [s[0] for s in starts]
        indices = [s[1] for s in starts]

        ans = [-1] * n
        for i in range(n):
            val = intervals[i][1]
            j = bisect.bisect_left(newstarts, val)
            if j < n:
                ans[i] = indices[j]
        return ans
